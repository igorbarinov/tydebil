Я придумал, как проще чем сечас, обрабатывать клики для множества обектов,
кода один накладывается на другой. Расмотрим пример фон и кнопка.(см. s16l43)

local flag = false -- показывает был ли клик по объекту до фона

local function done1(event) -- функция обрабатывающая клик по кнопке
if flag == false then   -- добавляем условие 
  flag = true  -- показываем что клик произошел
     ------- код обработки клика по кнопке------------
          if event.phase == "ended" then 
             print ("button") 
          end
     ------- код обработки клика по кнопке------------
end
end

local function done() -- функция обрабатывающая клик по фону
   if flag==false then -- проверяем а был ли ранее клик
 ------- код обработки клика по фону------------
       if event.phase == "ended" then
          print ("background") 
       end
 ------- код обработки клика по фону------------
   end  
 flag = false -- сбрасываем флаг. Данный код сработает при любом клике
end 

background:addEventListener ( "touch", done )
button:addEventListener ( "touch", done1 )

Суть в том что,событие клика по фону сработает при любом клике. И всегда это событие будет происходить поседним.
И в конц каждого клика будет происходить сброс флага.{ flag = false -- сбрасываем флаг.}
